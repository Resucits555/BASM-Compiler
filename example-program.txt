section .rdata
	qword msg = "hello, world!"


section .text
	extern GetStdHandle
	extern WriteConsoleA


	global main:
	mov rcx, msg ; buffer
	mov edx, 13  ; msglen
	call rel Print
	retn


	static Print:
	push edx
	push rcx
	mov ecx, -11 ; getting STDOUT
	call rel GetStdHandle

	mov rcx, rax ; STDOUT
	pop rdx      ; buffer
	pop r8d      ; to write (msglen)
	mov r9d, 0   ; not used (characters written)
	call rel WriteConsoleA
	retn
